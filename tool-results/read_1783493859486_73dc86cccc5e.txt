    80→export function VideoPreview() {
    81→  const ayatList = useBuilderStore((s) => s.ayatList)
    82→  const loading = useBuilderStore((s) => s.loadingAyats)
    83→  const ayatError = useBuilderStore((s) => s.ayatError)
    84→  const surahs = useBuilderStore((s) => s.surahs)
    85→  const selectedSurahNumber = useBuilderStore((s) => s.selectedSurahNumber)
    86→  const reciterId = useBuilderStore((s) => s.reciterId)
    87→  const translationKey = useBuilderStore((s) => s.translationKey)
    88→  const settings = useBuilderStore((s) => s.settings)
    89→  const fromAyat = useBuilderStore((s) => s.fromAyat)
    90→  const toAyat = useBuilderStore((s) => s.toAyat)
    91→
    92→  // Derive surah + reciter from raw state with stable references.
    93→  const surah = useMemo(
    94→    () => surahs.find((s) => s.number === selectedSurahNumber),
    95→    [surahs, selectedSurahNumber],
    96→  )
    97→  const reciter = useMemo(
    98→    () => RECITERS_LIST.find((r) => r.id === reciterId) ?? RECITERS_LIST[0]!,
    99→    [reciterId],
   100→  )
   101→
   102→  const audioRef = useRef<HTMLAudioElement | null>(null)
   103→  const rafRef = useRef<number | null>(null)
   104→  // Preloader: a hidden Audio() object that fetches + decodes the NEXT
   105→  // ayat's MP3 while the current one plays. When `onEnded` fires, we
   106→  // swap the preloaded src into the main audio element so playback
   107→  // resumes within ~50ms instead of the ~1000ms gap caused by fetching
   108→  // + decoding on demand.
   109→  const nextAudioRef = useRef<HTMLAudioElement | null>(null)
   110→
   111→  const [currentIndex, setCurrentIndex] = useState(0)
   112→  const [isPlaying, setIsPlaying] = useState(false)
   113→  const [currentTimeMs, setCurrentTimeMs] = useState(0)
   114→  const [liveDurations, setLiveDurations] = useState<Record<number, number>>({})
   115→  const [volume, setVolume] = useState(0.9)
   116→  const [muted, setMuted] = useState(false)
   117→
   118→  const current = ayatList[currentIndex]
   119→
   120→  // Attribution line for the selected translation edition. Empty for
   121→  // public-domain editions (no attribution needed). For permissive/personal
   122→  // editions, shown at the bottom-left of the preview + export.
   123→  const attributionLine = useMemo(
   124→    () => videoAttributionLine(translationKey),
   125→    [translationKey],
   126→  )
   127→
   128→  // Detect if the selected translation is Bengali — use the Bengali font
   129→  // for the translation text in the preview.
   130→  const isBengaliTranslation = translationKey.startsWith('bn.')
   131→
   132→  const aspect = ASPECT[settings.orientation]
   133→
   134→  // Total duration — use the probed audioDurationMs, or fall back to the
   135→  // live duration captured from the actual <audio> element's loadedmetadata.
   136→  // This handles the case where the probe returned 0 (COEP blocking the
   137→  // separate Audio() element) but the real audio element in the page loads fine.
   138→  const totalMs = useMemo(
   139→    () =>
   140→      ayatList.reduce(
   141→        (sum, a, i) =>
   142→          sum + (a.audioDurationMs || liveDurations[i] || 0),
   143→        0,
   144→      ),
   145→    [ayatList, liveDurations],
   146→  )
   147→
   148→  // Cumulative offsets so the seek bar spans the whole range.
   149→  const offsets = useMemo(() => {
   150→    const arr: number[] = []
   151→    let acc = 0
   152→    for (let i = 0; i < ayatList.length; i++) {
   153→      arr.push(acc)
   154→      acc += ayatList[i]?.audioDurationMs || liveDurations[i] || 0
   155→    }
   156→    return arr
   157→  }, [ayatList, liveDurations])
   158→
   159→  // -- per-ayat word highlight, driven by audio.currentTime -------------
   160→  const [activeWord, setActiveWord] = useState<ActiveWord | null>(null)
   161→
   162→  // Use a ref to break the recursive self-reference inside `tick` — this
   163→  // also lets us always read the latest `current`/`currentIndex` without
   164→  // re-creating the rAF loop on every state change.
   165→  const stateRef = useRef({ current, currentIndex })
   166→  useEffect(() => {
   167→    stateRef.current = { current, currentIndex }
   168→  }, [current, currentIndex])
   169→
   170→  useEffect(() => {
   171→    if (!isPlaying) return
   172→    let cancelled = false
   173→
   174→    const tick = () => {
   175→      if (cancelled) return
   176→      const audio = audioRef.current
   177→      const { current: c, currentIndex: ci } = stateRef.current
   178→      if (audio && c) {
   179→        const tMs = audio.currentTime * 1000
   180→        setCurrentTimeMs(tMs)
   181→        const foundIdx = getActiveWordIndex(c.words, tMs)
   182→        if (foundIdx >= 0) {
   183→          setActiveWord({ ayatIndex: ci, wordIndex: foundIdx })
   184→        }
   185→      }
   186→      rafRef.current = requestAnimationFrame(tick)
   187→    }
   188→    rafRef.current = requestAnimationFrame(tick)
   189→    return () => {
   190→      cancelled = true
   191→      if (rafRef.current) cancelAnimationFrame(rafRef.current)
   192→    }
   193→  }, [isPlaying])
   194→
   195→  // -- playback controls -------------------------------------------------
   196→  const playAyat = useCallback(
   197→    (idx: number) => {
   198→      if (!ayatList[idx]) return
   199→      const audio = audioRef.current
   200→      if (!audio) return
   201→      setCurrentIndex(idx)
   202→      setActiveWord(null)
   203→      setCurrentTimeMs(0)
   204→      audio.src = ayatList[idx]!.audioUrl
   205→      audio.volume = volume
   206→      audio.muted = muted
   207→      audio.play().then(() => setIsPlaying(true)).catch(() => {
   208→        // Autoplay can fail; show paused state and let the user retry.
   209→        setIsPlaying(false)
   210→      })
   211→    },
   212→    [ayatList, volume, muted],
   213→  )
   214→
   215→  // ─── Preload the NEXT ayat's MP3 in parallel ─────────────────────────
   216→  // Whenever currentIndex changes, kick off a hidden Audio() download of
   217→  // the next ayat's MP3. The browser fetches + decodes it in the
   218→  // background. When the current ayat ends, we swap the preloaded src
   219→  // into the main audio element — playback resumes in ~50ms instead of
   220→  // the ~1000ms gap caused by fetching on demand.
   221→  useEffect(() => {
   222→    if (!ayatList.length) return
   223→    const nextIdx = currentIndex + 1
   224→    if (nextIdx >= ayatList.length) return // last ayat — nothing to preload
   225→    const nextUrl = ayatList[nextIdx]?.audioUrl
   226→    if (!nextUrl) return
   227→
   228→    // Reuse the same hidden Audio() instance, just swap the src.
   229→    // `preload='auto'` tells the browser to fetch + buffer aggressively.
   230→    if (!nextAudioRef.current) {
   231→      nextAudioRef.current = new Audio()
   232→    }
   233→    const nextAudio = nextAudioRef.current
   234→    // Skip if already preloading/loaded this exact URL (avoid re-fetch).
   235→    if (nextAudio.src !== nextUrl) {
   236→      nextAudio.src = nextUrl
   237→      nextAudio.preload = 'auto'
   238→      // Some browsers won't actually fetch until play() is called.
   239→      // Calling play() then immediately pausing forces the preload
   240→      // without producing audible sound.
   241→      nextAudio.muted = true
   242→      nextAudio.volume = 0
   243→      nextAudio.play().then(() => {
   244→        nextAudio.pause()
   245→        nextAudio.currentTime = 0
   246→        nextAudio.muted = false
   247→        nextAudio.volume = 1
   248→      }).catch(() => {
   249→        // Autoplay policy blocked the preload — fall back to setting
   250→        // preload='auto' and letting the browser fetch metadata-only.
   251→        nextAudio.muted = false
   252→        nextAudio.volume = 1
   253→      })
   254→    }
   255→  }, [currentIndex, ayatList])
   256→
   257→  // Cleanup: release the preloader Audio() when the component unmounts
   258→  // so it doesn't keep a network connection open in the background.
   259→  useEffect(() => {
   260→    return () => {
   261→      if (nextAudioRef.current) {
   262→        nextAudioRef.current.src = ''
   263→        nextAudioRef.current = null
   264→      }
   265→    }
   266→  }, [])
   267→
   268→  // Auto-advance on ended
   269→  useEffect(() => {
   270→    const audio = audioRef.current
   271→    if (!audio) return
   272→    const onEnded = () => {
   273→      if (currentIndex < ayatList.length - 1) {
   274→        playAyat(currentIndex + 1)
   275→      } else {
   276→        // Last ayat just finished — pin the seek bar to the very end so it
   277→        // visually shows 100% completion (instead of jumping back to the
   278→        // start of the last ayat, which is what resetting currentTimeMs to
   279→        // 0 would do).
   280→        setIsPlaying(false)
   281→        setActiveWord(null)
   282→        const lastAyat = ayatList[currentIndex]
   283→        const lastDur = lastAyat?.audioDurationMs ?? 0
   284→        setCurrentTimeMs(lastDur)
   285→      }
   286→    }
   287→    audio.addEventListener('ended', onEnded)
   288→
   289→    // Capture the duration from the real audio element when it loads.
   290→    // This is the fallback for when the probe (getAudioDurationMs) returned 0
   291→    // because COEP blocked the separate Audio() element.
   292→    const onMeta = () => {
   293→      const d = audio.duration
   294→      if (isFinite(d) && d > 0) {
   295→        setLiveDurations((prev) => {
   296→          if (prev[currentIndex] === d * 1000) return prev
   297→          return { ...prev, [currentIndex]: d * 1000 }
   298→        })
   299→      }
   300→    }
   301→    audio.addEventListener('loadedmetadata', onMeta)
   302→    audio.addEventListener('durationchange', onMeta)
   303→
   304→    return () => {
   305→      audio.removeEventListener('ended', onEnded)
   306→      audio.removeEventListener('loadedmetadata', onMeta)
   307→      audio.removeEventListener('durationchange', onMeta)
   308→    }
   309→  }, [currentIndex, ayatList.length, ayatList, playAyat])
   310→
   311→  const togglePlay = () => {
   312→    const audio = audioRef.current
   313→    if (!audio) return
   314→    if (!ayatList.length) return
   315→    if (isPlaying) {
   316→      audio.pause()
   317→      setIsPlaying(false)
   318→    } else {
   319→      // If audio is empty (first play), load current ayat
   320→      if (!audio.src || audio.src === window.location.href) {
   321→        playAyat(currentIndex)
   322→      } else {
   323→        audio.play().then(() => setIsPlaying(true)).catch(() => setIsPlaying(false))
   324→      }
   325→    }
   326→  }
   327→
   328→  const onSeek = (v: number[]) => {
   329→    const target = v[0]!
   330→    // find which ayat this falls into
   331→    let idx = 0
   332→    for (let i = 0; i < offsets.length; i++) {
   333→      const start = offsets[i]!
   334→      const end = start + (ayatList[i]?.audioDurationMs || 0)
   335→      if (target >= start && target < end) {
   336→        idx = i
   337→        break
   338→      }
   339→      if (i === offsets.length - 1) idx = i
   340→    }
   341→    const into = target - (offsets[idx] || 0)
   342→    if (idx !== currentIndex) {
   343→      setCurrentIndex(idx)
   344→      const audio = audioRef.current!
   345→      audio.src = ayatList[idx]!.audioUrl
   346→      audio.currentTime = into / 1000
   347→      audio.volume = volume
   348→      audio.muted = muted
   349→      if (isPlaying) audio.play().catch(() => {})
   350→    } else {
   351→      const audio = audioRef.current!
   352→      audio.currentTime = into / 1000
   353→    }
   354→    setCurrentTimeMs(into)
   355→  }
   356→
   357→  useEffect(() => {
   358→    if (audioRef.current) {
   359→      audioRef.current.volume = volume
   360→      audioRef.current.muted = muted
   361→    }
   362→  }, [volume, muted])
   363→
   364→  // Reset when the ayat list changes. This is the documented React
   365→  // "reset all state when a prop changes" pattern — we can't use a `key`
   366→  // here because the player owns the audio element. The setState calls
   367→  // only fire when ayatList identity changes, so there's no cascading.
   368→  const prevListRef = useRef(ayatList)
   369→  useEffect(() => {
   370→    if (prevListRef.current === ayatList) return
   371→    prevListRef.current = ayatList
   372→    // eslint-disable-next-line react-hooks/set-state-in-effect
   373→    setCurrentIndex(0)
   374→    setIsPlaying(false)
   375→    setActiveWord(null)
   376→    setCurrentTimeMs(0)
   377→    setLiveDurations({})
   378→    if (audioRef.current) {
   379→      audioRef.current.pause()
   380→      audioRef.current.removeAttribute('src')
   381→    }
   382→  }, [ayatList])
   383→
   384→  // Pre-probe audio durations for all ayats when the list loads.
   385→  // The store's getAudioDurationMs may return 0 under COEP — this is a
   386→  // second attempt using the actual audio element in the DOM. We load each
   387→  // ayat's audio URL silently (muted, not playing) just to get the metadata.
   388→  useEffect(() => {
   389→    if (!ayatList.length) return
   390→    const probes: HTMLAudioElement[] = []
   391→    ayatList.forEach((ayat, i) => {
   392→      if (ayat.audioDurationMs > 0) return // already have it
   393→      const a = new Audio()
   394→      a.preload = 'metadata'
   395→      a.crossOrigin = 'anonymous'
   396→      a.onloadedmetadata = () => {
   397→        const d = a.duration
   398→        if (isFinite(d) && d > 0) {
   399→          setLiveDurations((prev) => ({ ...prev, [i]: d * 1000 }))
   400→        }
   401→      }
   402→      a.src = ayat.audioUrl
   403→      probes.push(a)
   404→    })
   405→    return () => {
   406→      probes.forEach((a) => { a.src = '' })
   407→    }
   408→  }, [ayatList])
   409→
   410→  // Build the overlay background expression from the user's chosen preset.
   411→  const overlayBg = overlayCssBackground(settings)
   412→
   413→  // Find the word to highlight in the current ayat
   414→  const highlightedWordIdx =
   415→    activeWord && activeWord.ayatIndex === currentIndex
   416→      ? activeWord.wordIndex
   417→      : -1
   418→
   419→  // Font sizes scale with the ACTUAL preview frame width using CSS container
   420→  // query units (cqw = 1% of the container's inline size). This way, the text
   421→  // is always proportional to the preview, not the browser viewport — so a
   422→  // narrow portrait reel gets smaller text and a wide landscape preview gets
   423→  // larger text, automatically.
   424→  //
   425→  // The user's font-size slider acts as a multiplier on top of the base
   426→  // percentage. Reference sizes match AUTO_FONT_SIZES.
   427→  const orientationFontBase: Record<string, { ar: number; tr: number; arRef: number; trRef: number }> = {
   428→    portrait: { ar: 7.0, tr: 2.8, arRef: 30, trRef: 14 },
   429→    landscape: { ar: 4.5, tr: 1.8, arRef: 34, trRef: 15 },
   430→  }
   431→  const fb = orientationFontBase[settings.orientation]!
   432→  const arCqw = (fb.ar * settings.arabicFontSize / fb.arRef).toFixed(2)
   433→  const trCqw = (fb.tr * settings.translationFontSize / fb.trRef).toFixed(2)
   434→  const arabicFontSizeCss = `${arCqw}cqw`
   435→  const translationFontSizeCss = `${trCqw}cqw`
   436→
   437→  return (
   438→    <div className="flex flex-col h-full">
   439→      {/* Hidden audio element drives playback + word timing */}
   440→      <audio ref={audioRef} preload="auto" crossOrigin="anonymous" />
   441→
   442→      {/* Preview frame — sized so portrait fills the available height (reel-like)
   443→          and landscape fills the available width. The frame is also a CSS
   444→          container so cqw units inside it scale with its actual width. */}
   445→      <div className="flex-1 min-h-0 grid place-items-center p-3 sm:p-6">
   446→        <div
   447→          className="qv-smooth relative overflow-hidden rounded-2xl shadow-2xl ring-1 ring-white/10"
   448→          style={{
   449→            aspectRatio: aspect.ratio,
   450→            // Portrait: fill the height, derive width from aspect-ratio.
   451→            // Landscape: fill the width, cap height so it never overflows.
   452→            ...(settings.orientation === 'portrait'
   453→              ? { height: '100%', width: 'auto', maxWidth: '100%' }
   454→              : { width: '100%', maxHeight: '100%', height: 'auto' }),
   455→            containerType: 'inline-size',
   456→            backgroundImage: `url(${settings.backgroundImage})`,
   457→            backgroundSize: 'cover',
   458→            backgroundPosition: 'center',
   459→          }}
   460→        >
   461→          {/* Overlay (rendered using the user's selected preset shape) */}
   462→          {overlayBg && (
   463→            <div
   464→              className="absolute inset-0 pointer-events-none"
   465→              style={{ background: overlayBg }}
   466→            />
   467→          )}
   468→
   469→          {/* Subtle top + bottom gradient for legibility (always on, regardless
   470→              of overlay preset — guarantees the header + watermark stay readable) */}
   471→          <div
   472→            className="absolute inset-0 pointer-events-none"
   473→            style={{
   474→              background:
   475→                'linear-gradient(180deg, rgba(0,0,0,0.22) 0%, rgba(0,0,0,0) 18%, rgba(0,0,0,0) 78%, rgba(0,0,0,0.30) 100%)',
   476→            }}
   477→          />
   478→
   479→          {/* Top header bar — surah name + ayat indicator (scales with frame) */}
   480→          {surah && current && (
   481→            <div
   482→              className="absolute top-0 inset-x-0 flex items-start justify-between text-white"
   483→              style={{ padding: '4cqw 5cqw' }}
   484→            >
   485→              <div className="flex flex-col">
   486→                <span
   487→                  lang="ar"
   488→                  className="font-arabic-uthmani leading-tight drop-shadow-lg"
   489→                  style={{ fontSize: '5cqw' }}
   490→                >
   491→                  {surah.arabicName}
   492→                </span>
   493→                <span
   494→                  className="uppercase tracking-[0.12em] opacity-75 mt-0.5"
   495→                  style={{ fontSize: '2cqw' }}
   496→                >
   497→                  {surah.name} · {surah.revelationType}
   498→                </span>
   499→              </div>
   500→              <div className="flex flex-col items-end">
   501→                <span
   502→                  className="font-arabic-uthmani leading-tight drop-shadow-lg"
   503→                  style={{ fontSize: '4.2cqw' }}
   504→                >
   505→                  {surah.number}:{current.ayatNumber}
   506→                </span>
   507→                <span
   508→                  className="uppercase tracking-[0.18em] opacity-65 mt-0.5"
   509→                  style={{ fontSize: '1.8cqw' }}
   510→                >
   511→                  Ayat {currentIndex + 1} of {ayatList.length}
   512→                </span>
   513→              </div>
   514→            </div>
   515→          )}
   516→
   517→          {/* Center content — Arabic + transliteration + translation grouped
   518→              tightly into one visual block. No card border (per spec).
   519→              Padding scales with the frame so it stays proportional. */}
   520→          {current ? (
   521→            <div
   522→              className="absolute inset-0 flex flex-col items-center justify-center"
   523→              style={{ padding: '0 8cqw' }}
   524→            >
   525→              <div
   526→                className="qv-smooth relative flex flex-col items-center rounded-2xl"
   527→                style={{
   528→                  maxWidth: TEXT_WIDTH_MAP[settings.textWidth],
   529→                  backgroundColor: 'rgba(15, 15, 20, 0.6)',
   530→                  padding: '4cqw 5cqw',
   531→                  borderRadius: '3cqw',
   532→                  boxShadow: '0 2cqw 6cqw rgba(0, 0, 0, 0.4)',
   533→                }}
   534→              >
   535→                {/* Arabic — word-by-word highlight */}
   536→                <div
   537→                  dir="rtl"
   538→                  lang="ar"
   539→                  className={cn(
   540→                    'qv-smooth text-center leading-[1.75] drop-shadow-lg',
   541→                    ARABIC_FONT_CLASS[settings.arabicFont] ?? 'font-arabic-uthmani',
   542→                  )}
   543→                  style={{
   544→                    color: settings.fontColor,
   545→                    fontSize: arabicFontSizeCss,
   546→                  }}
   547→                >
   548→                  {current.words.length > 0
   549→                    ? current.words.map((w, i) => (
   550→                        <span
   551→                          key={i}
   552→                          className="qv-smooth inline-block mx-[1px]"
   553→                          style={{
   554→                            color:
   555→                              i === highlightedWordIdx
   556→                                ? settings.highlightColor
   557→                                : settings.fontColor,
   558→                            textShadow:
   559→                              i === highlightedWordIdx
   560→                                ? `0 0 18px ${settings.highlightColor}88, 0 1px 4px rgba(0,0,0,0.7)`
   561→                                : '0 1px 4px rgba(0,0,0,0.7)',
   562→                            transition: 'color 120ms ease',
   563→                          }}
   564→                        >
   565→                          {w.text}
   566→                        </span>
   567→                      ))
   568→                    : current.arabicText}
   569→                </div>
   570→
   571→                {/* Tiny divider between Arabic and translation — only when
   572→                    both are visible. */}
   573→                {settings.showTranslation &&
   574→                  settings.showTransliteration &&
   575→                  current.words.length > 0 && (
   576→                    <div
   577→                      className="my-2 h-px opacity-40"
   578→                      style={{ backgroundColor: settings.fontColor, width: '12cqw' }}
   579→                    />
   580→                  )}
   581→
   582→                {/* Transliteration — sits right under Arabic (scales with frame) */}
   583→                {settings.showTransliteration && current.words.length > 0 && (
   584→                  <div
   585→                    className="qv-smooth text-center italic text-white/70 leading-snug"
   586→                    style={{ fontSize: '2.4cqw', maxWidth: '80cqw' }}
   587→                  >
   588→                    {current.words
   589→                      .map((w) => w.transliteration || '')
   590→                      .filter(Boolean)
   591→                      .join(' ')}
   592→                  </div>
   593→                )}
   594→
   595→                {/* Translation — sits right under Arabic (or transliteration).
   596→                    Tight gap, scales with frame. Uses the selected Bengali
   597→                    font when the translation is Bengali, otherwise the
   598→                    default UI font (Inter). */}
   599→                {settings.showTranslation && (
   600→                  <p
   601→                    className={`qv-smooth text-white/85 mx-auto leading-snug drop-shadow text-center ${isBengaliTranslation ? (BENGALI_FONT_CLASS[settings.bengaliFont] ?? 'font-bengali-sans') : ''}`}
   602→                    style={{
   603→                      fontSize: translationFontSizeCss,
   604→                      marginTop: TEXT_SPACING_MAP[settings.textSpacing],
   605→                      maxWidth: '85cqw',
   606→                    }}
   607→                  >
   608→                    {current.translation}
   609→                  </p>
   610→                )}
   611→              </div>
   612→            </div>
   613→          ) : (
   614→            <div className="absolute inset-0 grid place-items-center text-center text-white/70 px-6">
   615→              {loading ? (
   616→                <div className="flex flex-col items-center gap-3">
   617→                  <Loader2 className="h-8 w-8 animate-spin text-primary" />
   618→                  <span className="text-sm">Loading ayat data…</span>
   619→                </div>
   620→              ) : ayatError ? (
   621→                <div className="space-y-3">
   622→                  <p className="text-sm text-destructive">{ayatError}</p>
   623→                </div>
   624→              ) : (
   625→                <div className="space-y-4 max-w-sm">
   626→                  <div className="grid place-items-center h-14 w-14 rounded-2xl bg-primary/10 ring-1 ring-primary/20 mx-auto">
   627→                    <Sparkles className="h-7 w-7 text-primary" />
   628→                  </div>
   629→                  <div className="space-y-1.5">
   630→                    <p className="text-sm font-medium text-white/90">
   631→                      Your recitation video will appear here
   632→                    </p>
   633→                    <p className="text-xs text-white/55 leading-relaxed">
   634→                      Pick a surah and ayat range, choose a reciter, then click{' '}
   635→                      <span className="text-primary font-medium">Load ayats</span>{' '}
   636→                      to preview with word-by-word highlighting.
   637→                    </p>
   638→                  </div>
   639→                </div>
   640→              )}
   641→            </div>
   642→          )}
   643→
   644→          {/* Attribution block — bottom-left. Shows the translation attribution
   645→              line (when the edition requires it) + the reciter credit (always).
   646→              Stacked vertically so both are visible. */}
   647→          {(attributionLine || ayatList.length > 0) && (
   648→            <div
   649→              className="absolute text-white/55 font-sans leading-tight max-w-[55%] space-y-0.5"
   650→              style={{
   651→                bottom: '2.5cqw',
   652→                left: '3.5cqw',
   653→                fontSize: '2.4cqw',
   654→              }}
   655→            >
   656→              {attributionLine && <div>{attributionLine}</div>}
   657→              {ayatList.length > 0 && (
   658→                <div>Recited by {reciter.name}</div>
   659→              )}
   660→            </div>
   661→          )}
   662→
   663→          {/* Structural Quran markers — Juz, Hizb, Rubʿ al-Hizb, Ruku,
   664→              Manzil, Page. Sourced from the quran.com API. Shown at the
   665→              bottom-center of the frame, just above the attribution block,
   666→              so users can see which part of the Quran the current ayat
   667→              belongs to. Only shows fields that are actually present. */}
   668→          {current && hasStructuralInfo(current) && (
   669→            <div
   670→              className="absolute left-1/2 -translate-x-1/2 text-white/55 font-sans tracking-[0.08em] whitespace-nowrap"
   671→              style={{
   672→                bottom: '2.8cqw',
   673→                fontSize: '1.7cqw',
   674→                textShadow: '0 1px 4px rgba(0,0,0,0.6)',
   675→              }}
   676→            >
   677→              {formatStructural(current, false)}
   678→            </div>
   679→          )}
   680→
   681→          {/* Top-center watermark — the Jariyah Now brand mark (cleaned
   682→              transparent PNG at /public/watermark.png). Mirrors what
   683→              gets baked into the exported MP4 by ExportModal.drawFrame,
   684→              so the user sees the same branding in the live preview as
   685→              they will in the final video. */}
   686→          <img
   687→            src="/watermark.png"
   688→            alt=""
   689→            aria-hidden="true"
   690→            className="absolute pointer-events-none select-none"
   691→            style={{
   692→              top: '4cqw',
   693→              left: '50%',
   694→              transform: 'translateX(-50%)',
   695→              height: '14cqw',
   696→              width: 'auto',
   697→              opacity: 0.9,
   698→              filter: 'drop-shadow(0 1px 4px rgba(0,0,0,0.6))',
   699→            }}
   700→          />
   701→        </div>
   702→      </div>
   703→
   704→      {/* Controls bar — light theme, responsive: 1 row on desktop, 2 rows on mobile */}
   705→      <div className="border-t border-border bg-card px-3 sm:px-5 py-2.5 sm:py-3 space-y-2">
   706→        {/* Row 1: transport buttons + seek bar + volume (desktop) / transport + seek (mobile) */}
   707→        <div className="flex items-center gap-2 sm:gap-2.5">
   708→          <div className="flex items-center gap-1.5 sm:gap-2.5 shrink-0">
   709→            <Button
   710→              variant="ghost"
   711→              size="icon"
   712→              className="h-9 w-9 rounded-lg hover:bg-muted"
   713→              disabled={!ayatList.length}
   714→              onClick={() => playAyat(Math.max(0, currentIndex - 1))}
   715→              title="Previous ayat"
   716→              aria-label="Previous ayat"
   717→            >
   718→              <SkipBack className="h-4 w-4" />
   719→            </Button>
   720→            <Button
   721→              size="icon"
   722→              className="h-11 w-11 rounded-full qv-btn-primary"
   723→              disabled={!ayatList.length}
   724→              onClick={togglePlay}
   725→              title={isPlaying ? 'Pause' : 'Play'}
   726→              aria-label={isPlaying ? 'Pause' : 'Play'}
   727→            >
   728→              {isPlaying ? (
   729→                <Pause className="h-4 w-4" />
   730→              ) : (
   731→                <Play className="h-4 w-4 translate-x-0.5" />
   732→              )}
   733→            </Button>
   734→            <Button
   735→              variant="ghost"
   736→              size="icon"
   737→              className="h-9 w-9 rounded-lg hover:bg-muted"
   738→              disabled={!ayatList.length || currentIndex >= ayatList.length - 1}
   739→              onClick={() => playAyat(Math.min(ayatList.length - 1, currentIndex + 1))}
   740→              title="Next ayat"
   741→              aria-label="Next ayat"
   742→            >
   743→              <SkipForward className="h-4 w-4" />
   744→            </Button>
   745→          </div>
   746→
   747→          {/* Seek bar — flex-1 so it fills available space on all screens */}
   748→          <div className="flex-1 flex items-center gap-2 sm:gap-3 min-w-0">
   749→            <Slider
   750→              value={[
   751→                Math.min(
   752→                  totalMs,
   753→                  (offsets[currentIndex] || 0) + currentTimeMs,
   754→                ),
   755→              ]}
   756→              max={Math.max(1, totalMs)}
   757→              step={100}
   758→              onValueChange={onSeek}
   759→              disabled={!ayatList.length}
   760→              className="flex-1 min-w-0"
   761→            />
   762→            <span className="text-[11px] font-mono text-muted-foreground whitespace-nowrap tabular-nums shrink-0">
   763→              {formatMs((offsets[currentIndex] || 0) + currentTimeMs)} /{' '}
   764→              {formatMs(totalMs)}
   765→            </span>
   766→          </div>
   767→
   768→          {/* Volume — hidden on mobile (use device volume), shown on sm+ */}
   769→          <div className="hidden sm:flex items-center gap-1.5 shrink-0">
   770→            <Button
   771→              variant="ghost"
   772→              size="icon"
   773→              className="h-9 w-9 rounded-lg hover:bg-muted"
   774→              onClick={() => setMuted((m) => !m)}
   775→              title={muted ? 'Unmute' : 'Mute'}
   776→              aria-label={muted ? 'Unmute' : 'Mute'}
   777→            >
   778→              {muted || volume === 0 ? (
   779→                <VolumeX className="h-4 w-4" />
   780→              ) : (
   781→                <Volume2 className="h-4 w-4" />
   782→              )}
   783→            </Button>
   784→            <Slider
   785→              value={[muted ? 0 : volume * 100]}
   786→              max={100}
   787→              step={1}
   788→              onValueChange={(v) => {
   789→                setVolume(v[0]! / 100)
   790→                setMuted(false)
   791→              }}
   792→              className="w-20"
   793→            />
   794→          </div>
   795→
   796→          {/* Mute-only on mobile */}
   797→          <Button
   798→            variant="ghost"
   799→            size="icon"
   800→            className="sm:hidden h-9 w-9 rounded-lg hover:bg-muted shrink-0"
   801→            onClick={() => setMuted((m) => !m)}
   802→            title={muted ? 'Unmute' : 'Mute'}
   803→            aria-label={muted ? 'Unmute' : 'Mute'}
   804→          >
   805→            {muted || volume === 0 ? (
   806→              <VolumeX className="h-4 w-4" />
   807→            ) : (
   808→              <Volume2 className="h-4 w-4" />
   809→            )}
   810→          </Button>
   811→        </div>
   812→
   813→        {/* Row 2: status line */}
   814→        {ayatList.length > 0 && (
   815→          <div className="flex items-center justify-between text-[11px] text-muted-foreground pt-0.5">
   816→            <span className="flex items-center gap-2 min-w-0">
   817→              <span className="font-medium text-foreground/80 truncate">{surah?.name}</span>
   818→              <span className="opacity-50 shrink-0">·</span>
   819→              <span className="tabular-nums shrink-0">
   820→                Ayats {fromAyat}–{toAyat}
   821→              </span>
   822→              <span className="opacity-50 shrink-0 hidden xs:inline">·</span>
   823→              <span className="hidden xs:inline truncate">{reciter.name}</span>
   824→            </span>
   825→            <span className="flex items-center gap-1.5 text-foreground/70 shrink-0">
   826→              <span className="inline-block h-1.5 w-1.5 rounded-full bg-primary qv-pulse" />
   827→              <span className="uppercase tracking-[0.15em] text-[10px]">
   828→                Live preview
   829→              </span>
   830→            </span>
   831→          </div>
   832→        )}
   833→      </div>
   834→    </div>
   835→  )
   836→}
   837→
   838→function formatMs(ms: number): string {
   839→  if (!Number.isFinite(ms) || ms <= 0) return '0:00'
   840→  const total = Math.floor(ms / 1000)
   841→  const m = Math.floor(total / 60)
   842→  const s = total % 60
   843→  return `${m}:${String(s).padStart(2, '0')}`
   844→}
