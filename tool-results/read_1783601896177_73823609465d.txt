    67→    67→/**
    68→    68→ * Custom video preview player with proper controls (play/pause, seek bar,
    69→    69→ * time display, volume). Replaces the bare <video controls> which showed
    70→    70→ * 0:00 / 0:00 until metadata loaded and had no visible duration.

    72→    72→function VideoPreviewPlayer({
    73→    73→  src,
    74→    74→  orientation,
    75→    75→  filename,
    76→    76→  isMp4,
    77→    77→}: {
    78→    78→  src: string
    79→    79→  orientation: Orientation
    80→    80→  filename: string
    81→    81→  isMp4: boolean
    82→    82→}) {
    83→    83→  const videoRef = useRef<HTMLVideoElement | null>(null)
    84→    84→  const [isPlaying, setIsPlaying] = useState(false)
    85→    85→  const [currentTime, setCurrentTime] = useState(0)
    86→    86→  const [duration, setDuration] = useState(0)
    87→    87→  const [volume, setVolume] = useState(1)
    88→    88→  const [muted, setMuted] = useState(false)
    89→    89→
    90→    90→  useEffect(() => {
    91→    91→    const v = videoRef.current
    92→    92→    if (!v) return
    93→    93→    const onTime = () => setCurrentTime(v.currentTime)
    94→    94→    const onDur = () => setDuration(v.duration || 0)
    95→    95→    const onPlay = () => setIsPlaying(true)
    96→    96→    const onPause = () => setIsPlaying(false)
    97→    97→    v.addEventListener('timeupdate', onTime)
    98→    98→    v.addEventListener('loadedmetadata', onDur)
    99→    99→    v.addEventListener('durationchange', onDur)
   100→   100→    v.addEventListener('play', onPlay)
   101→   101→    v.addEventListener('pause', onPause)
   102→   102→    // Auto-play muted on load so the user sees motion immediately
   103→   103→    v.muted = true
   104→   104→    v.play().catch(() => {})
   105→   105→    return () => {
   106→   106→      v.removeEventListener('timeupdate', onTime)
   107→   107→      v.removeEventListener('loadedmetadata', onDur)
   108→   108→      v.removeEventListener('durationchange', onDur)
   109→   109→      v.removeEventListener('play', onPlay)
   110→   110→      v.removeEventListener('pause', onPause)
   111→   111→    }
   112→   112→  }, [src])
   113→   113→
   114→   114→  const togglePlay = () => {
   115→   115→    const v = videoRef.current
   116→   116→    if (!v) return
   117→   117→    if (v.paused) v.play()
   118→   118→    else v.pause()
   119→   119→  }
   120→   120→
   121→   121→  const onSeek = (val: number[]) => {
   122→   122→    const v = videoRef.current
   123→   123→    if (!v || !duration) return
   124→   124→    v.currentTime = (val[0]! / 100) * duration
   125→   125→    setCurrentTime(v.currentTime)
   126→   126→  }
   127→   127→
   128→   128→  const onVol = (val: number[]) => {
   129→   129→    const v = videoRef.current
   130→   130→    if (!v) return
   131→   131→    const vol = val[0]! / 100
   132→   132→    v.volume = vol
   133→   133→    v.muted = vol === 0
   134→   134→    setVolume(vol)
   135→   135→    setMuted(vol === 0)
   136→   136→  }
   137→   137→
   138→   138→  const toggleMute = () => {
   139→   139→    const v = videoRef.current
   140→   140→    if (!v) return
   141→   141→    v.muted = !v.muted
   142→   142→    setMuted(v.muted)
   143→   143→  }
   144→   144→
   145→   145→  const pct = duration > 0 ? (currentTime / duration) * 100 : 0
   146→   146→
   147→   147→  return (
   148→   148→    <div className="space-y-3">
   149→   149→      <div className="flex items-center gap-2 text-sm text-primary px-1">
   150→   150→        <CheckCircle2 className="h-5 w-5" />
   151→   151→        <span className="font-medium">Video ready!</span>
   152→   152→        <span className="text-xs text-muted-foreground ml-auto font-mono">
   153→   153→          {formatMs(duration * 1000)}
   154→   154→        </span>
   155→   155→      </div>
   156→   156→
   157→   157→      {/* Video + custom controls overlay */}
   158→   158→      <div className="relative rounded-xl overflow-hidden bg-black">
   159→   159→        <video
   160→   160→          ref={videoRef}
   161→   161→          preload="auto"
   162→   162→          playsInline
   163→   163→          loop
   164→   164→          onClick={togglePlay}
   165→   165→          className={`w-full block ${orientation === 'portrait' ? 'aspect-[9/16]' : 'aspect-video'}`}
   166→   166→        >
   167→   167→          <source
   168→   168→            src={src}
   169→   169→            type={
   170→   170→              isMp4
   171→   171→                ? 'video/mp4; codecs="avc1.42E01E, mp4a.40.2"'
   172→   172→                : 'video/webm'
   173→   173→            }
   174→   174→          />
   175→   175→        </video>
   176→   176→
   177→   177→        {/* Center play/pause button overlay */}
   178→   178→        {!isPlaying && (
   179→   179→          <button
   180→   180→            onClick={togglePlay}
   181→   181→            className="absolute inset-0 grid place-items-center bg-black/30 transition"
   182→   182→            aria-label="Play"
   183→   183→          >
   184→   184→            <div className="grid place-items-center h-14 w-14 rounded-full bg-primary/90 text-primary-foreground shadow-lg">
   185→   185→              <Play className="h-7 w-7 translate-x-0.5" />
   186→   186→            </div>
   187→   187→          </button>
   188→   188→        )}
   189→   189→      </div>
   190→   190→
   191→   191→      {/* Custom controls bar */}
   192→   192→      <div className="flex items-center gap-2 px-1">
   193→   193→        <button
   194→   194→          onClick={togglePlay}
   195→   195→          className="grid place-items-center h-8 w-8 rounded-lg hover:bg-muted shrink-0"
   196→   196→          aria-label={isPlaying ? 'Pause' : 'Play'}
   197→   197→        >
   198→   198→          {isPlaying ? <Pause className="h-4 w-4" /> : <Play className="h-4 w-4 translate-x-0.5" />}
   199→   199→        </button>
   200→   200→
   201→   201→        {/* Seek bar */}
   202→   202→        <Slider
   203→   203→          value={[pct]}
   204→   204→          max={100}
   205→   205→          step={0.1}
   206→   206→          onValueChange={onSeek}
   207→   207→          className="flex-1"
   208→   208→        />
   209→   209→
   210→   210→        {/* Time display */}
   211→   211→        <span className="text-xs font-mono tabular-nums text-muted-foreground whitespace-nowrap shrink-0">
   212→   212→          {formatMs(currentTime * 1000)} / {formatMs(duration * 1000)}
   213→   213→        </span>
   214→   214→
   215→   215→        {/* Volume */}
   216→   216→        <button
   217→   217→          onClick={toggleMute}
   218→   218→          className="grid place-items-center h-8 w-8 rounded-lg hover:bg-muted shrink-0"
   219→   219→          aria-label={muted ? 'Unmute' : 'Mute'}
   220→   220→        >
   221→   221→          {muted || volume === 0 ? <VolumeX className="h-4 w-4" /> : <Volume2 className="h-4 w-4" />}
   222→   222→        </button>
   223→   223→        <Slider
   224→   224→          value={[muted ? 0 : volume * 100]}
   225→   225→          max={100}
   226→   226→          step={1}
   227→   227→          onValueChange={onVol}
   228→   228→          className="w-16 shrink-0"
   229→   229→        />
   230→   230→      </div>
   231→   231→
   232→   232→      {/* Download button */}
   233→   233→      <a
   234→   234→        href={src}
   235→   235→        download={filename}
   236→   236→        className="inline-flex items-center justify-center gap-2 w-full h-11 rounded-xl qv-btn-primary text-sm font-semibold"
   237→   237→      >
   238→   238→        <Download className="h-4 w-4" />
   239→   239→        Download {filename}
   240→   240→      </a>
   241→   241→    </div>
   242→   242→  )
   243→   243→}

   977→   977→async function renderVideoToWebm({
   978→   978→  canvas,
   979→   979→  slides,
   980→   980→  settings,
   981→   981→  orientation,
   982→   982→  quality,
   983→   983→  attributionLine,
   984→   984→  reciterName,
   985→   985→  onProgress,
   986→   986→  activeSplit,
   987→   987→}: RenderArgs): Promise<{ url: string; blob: Blob }> {
   988→   988→  const base = RES[orientation]
   989→   989→  const scale = RENDER_QUALITY_SCALE[quality]
   990→   990→  const W = Math.round(base.w * scale)
   991→   991→  const H = Math.round(base.h * scale)
   992→   992→  canvas.width = W
   993→   993→  canvas.height = H
   994→   994→  const ctx = canvas.getContext('2d')!
   995→   995→  if (!ctx) throw new Error('Canvas 2D context unavailable')
   996→   996→
   997→   997→  // Load the background — for an MP4 video background we use a separate
   998→   998→  // <video> element that we'll seek frame-by-frame in the render loop;
   999→   999→  // for an image we use a regular <img>. Either way, `bgImg`/`bgVideo`
  1000→  1000→  // carries the loaded media (the other is null).
  1001→  1001→  const isVideoBg = settings.backgroundImage.endsWith('.mp4')
  1002→  1002→  const bgImg = isVideoBg
  1003→  1003→    ? null
  1004→  1004→    : await loadImage(settings.backgroundImage).catch(() => null)
  1005→  1005→  const bgVideo = isVideoBg
  1006→  1006→    ? await loadVideo(settings.backgroundImage).catch(() => null)
  1007→  1007→    : null
  1008→  1008→
  1009→  1009→  // Load the watermark image once.
  1010→  1010→  const watermarkImg = await loadImage('/watermark.png').catch(() => null)
  1011→  1011→
  1012→  1012→  // Pre-load all ayat audio as AudioBuffers via Web Audio API so we can
  1013→  1013→  // schedule them precisely on a single MediaRecorder timeline.
  1014→  1014→  // Safari < 14 ships AudioContext as webkitAudioContext. The capabilities
  1015→  1015→  // check at the top of the modal should prevent us reaching this code path
  1016→  1016→  // on a browser with neither variant, but we guard defensively anyway —
  1017→  1017→  // throwing a clear error is better than a non-null-assertion crash.
  1018→  1018→  type AudioContextCtor = typeof AudioContext
  1019→  1019→  const win = window as unknown as {
  1020→  1020→    AudioContext?: AudioContextCtor
  1021→  1021→    webkitAudioContext?: AudioContextCtor
  1022→  1022→  }
  1023→  1023→  const AudioCtx = win.AudioContext ?? win.webkitAudioContext
  1024→  1024→  if (!AudioCtx) {
  1025→  1025→    throw new Error(
  1026→  1026→      'Web Audio API is not available in this browser. Try Chrome, Edge, or Firefox on desktop.',
  1027→  1027→    )
  1028→  1028→  }
  1029→  1029→  const audioCtx = new AudioCtx()
  1030→  1030→  const buffers: AudioBuffer[] = []
  1031→  1031→  for (const s of slides) {
  1032→  1032→    const buf = await fetchAudioBuffer(s.audioUrl, audioCtx)
  1033→  1033→    buffers.push(buf)
  1034→  1034→  }
  1035→  1035→  const totalMs = slides.reduce(
  1036→  1036→    (sum, s, i) => sum + (buffers[i]?.duration ?? s.audioDurationMs / 1000) * 1000,
  1037→  1037→    0,
  1038→  1038→  )
  1039→  1039→  if (totalMs <= 0) {
  1040→  1040→    throw new Error('No audio could be loaded for the selected range.')
  1041→  1041→  }
  1042→  1042→
  1043→  1043→  // Set up MediaRecorder on a canvas.captureStream() track.
  1044→  1044→  const stream = canvas.captureStream(RENDER_FPS)
  1045→  1045→  // Create a MediaStreamDestination to mix the audio into the recording.
  1046→  1046→  // Try to add audio tracks, but don't fail if it doesn't work.
  1047→  1047→  let dest: MediaStreamAudioDestinationNode | null = null
  1048→  1048→  try {
  1049→  1049→    dest = audioCtx.createMediaStreamDestination()
  1050→  1050→    for (const tr of dest.stream.getAudioTracks()) {
  1051→  1051→      stream.addTrack(tr)
  1052→  1052→    }
  1053→  1053→  } catch {
  1054→  1054→    // Audio mixing failed — record video only (silent video)
  1055→  1055→  }
  1056→  1056→
  1057→  1057→  // Pick a supported MIME type. Uses the shared picker so the pre-flight
  1058→  1058→  // check and the actual render agree on what's supported.
  1059→  1059→  const mime = pickSupportedMimeType()
  1060→  1060→  const recorder = new MediaRecorder(
  1061→  1061→    stream,
  1062→  1062→    mime
  1063→  1063→      ? { mimeType: mime, videoBitsPerSecond: RENDER_VIDEO_BITRATE }
  1064→  1064→      : undefined,
  1065→  1065→  )
  1066→  1066→
  1067→  1067→  const chunks: BlobPart[] = []
  1068→  1068→  recorder.ondataavailable = (e) => {
  1069→  1069→    if (e.data.size > 0) chunks.push(e.data)
  1070→  1070→  }
  1071→  1071→
  1072→  1072→  const stopped = new Promise<void>((resolve) => {
  1073→  1073→    recorder.onstop = () => resolve()
  1074→  1074→  })
  1075→  1075→
  1076→  1076→  // Schedule all audio buffers back-to-back, starting a little ahead of "now"
  1077→  1077→  // so the first sample doesn't get clipped by the recorder's startup latency.
  1078→  1078→  const sources: AudioBufferSourceNode[] = []
  1079→  1079→  const leadSec = RENDER_AUDIO_LEAD_MS / 1000
  1080→  1080→  const startTime = audioCtx.currentTime + leadSec
  1081→  1081→  let t0 = startTime
  1082→  1082→  const startTimes: number[] = [] // audioCtx-time at which each ayat begins
  1083→  1083→  buffers.forEach((b) => {
  1084→  1084→    const src = audioCtx.createBufferSource()
  1085→  1085→    src.buffer = b
  1086→  1086→    if (dest) src.connect(dest)
  1087→  1087→    src.start(t0)
  1088→  1088→    startTimes.push(t0)
  1089→  1089→    sources.push(src)
  1090→  1090→    t0 += b.duration
  1091→  1091→  })
  1092→  1092→  const totalSec = t0 - startTime
  1093→  1093→
  1094→  1094→  recorder.start() // collect all data at stop() — faster than timeslice
  1095→  1095→
  1096→  1096→  // Drawing loop
  1097→  1097→  const startWall = performance.now()
  1098→  1098→  await new Promise<void>((resolve) => {
  1099→  1099→    const draw = () => {
  1100→  1100→      const elapsedSec = (performance.now() - startWall) / 1000
  1101→  1101→      // figure out which slide we're on
  1102→  1102→      let acc = 0
  1103→  1103→      let idx = 0
  1104→  1104→      let intoMs = 0
  1105→  1105→      for (let i = 0; i < slides.length; i++) {
  1106→  1106→        const dur = (buffers[i]?.duration ?? 0) * 1000
  1107→  1107→        if (elapsedSec * 1000 < acc + dur) {
  1108→  1108→          idx = i
  1109→  1109→          intoMs = elapsedSec * 1000 - acc
  1110→  1110→          break
  1111→  1111→        }
  1112→  1112→        acc += dur
  1113→  1113→        if (i === slides.length - 1) {
  1114→  1114→          idx = i
  1115→  1115→          intoMs = dur
  1116→  1116→        }
  1117→  1117→      }
  1118→  1118→
  1119→  1119→      // ── Seek the background video to the current frame ───────────────
  1120→  1120→      // The video loops every `bgDurationSec` seconds. We seek to
  1121→  1121→      // `elapsedSec % bgDurationSec` so the background animation stays
  1122→  1122→      // in sync with the recitation timeline. Seeking is async, but
  1123→  1123→      // for short clips (<10s) the browser usually has the frame ready
  1124→  1124→      // by the next rAF tick; we draw whatever frame is currently
  1125→  1125→      // visible (which may be the previous frame for one tick).
  1126→  1126→      if (bgVideo) {
  1127→  1127→        const bgDur = bgVideo.duration
  1128→  1128→        if (Number.isFinite(bgDur) && bgDur > 0) {
  1129→  1129→          const targetT = elapsedSec % bgDur
  1130→  1130→          // Only seek if we're more than ~1 frame off — avoids excessive
  1131→  1131→          // re-seeking that can stall the video element.
  1132→  1132→          if (Math.abs(bgVideo.currentTime - targetT) > 1 / RENDER_FPS) {
  1133→  1133→            try { bgVideo.currentTime = targetT } catch { /* seeking can throw if not ready */ }
  1134→  1134→          }
  1135→  1135→        }
  1136→  1136→      }
  1137→  1137→
  1138→  1138→      drawFrame({
  1139→  1139→        ctx,
  1140→  1140→        W,
  1141→  1141→        H,
  1142→  1142→        slide: slides[idx]!,
  1143→  1143→        intoMs,
  1144→  1144→        settings,
  1145→  1145→        bgImg,
  1146→  1146→        bgVideo,
  1147→  1147→        ayatIndex: idx,
  1148→  1148→        total: slides.length,
  1149→  1149→        attributionLine,
  1150→  1150→        reciterName,
  1151→  1151→        watermarkImg,
  1152→  1152→        splitPart: activeSplit && activeSplit.segments.length > 0
  1153→  1153→          ? { partNumber: idx + 1, totalParts: activeSplit.segments.length }
  1154→  1154→          : null,
  1155→  1155→      })
  1156→  1156→
  1157→  1157→      onProgress(Math.min(1, elapsedSec / totalSec))
  1158→  1158→
  1159→  1159→      if (elapsedSec >= totalSec) {
  1160→  1160→        resolve()
  1161→  1161→        return
  1162→  1162→      }
  1163→  1163→      requestAnimationFrame(draw)
  1164→  1164→    }
  1165→  1165→    requestAnimationFrame(draw)
  1166→  1166→  })
  1167→  1167→
  1168→  1168→  // Stop everything
  1169→  1169→  sources.forEach((s) => {
  1170→  1170→    try { s.stop() } catch {}
  1171→  1171→  })
  1172→  1172→  recorder.stop()
  1173→  1173→  await stopped
  1174→  1174→  audioCtx.close()
  1175→  1175→
  1176→  1176→  const blob = new Blob(chunks, { type: mime || 'video/webm' })
  1177→  1177→  const url = URL.createObjectURL(blob)
  1178→  1178→  return { url, blob }
  1179→  1179→}

  1181→  1181→function loadImage(src: string): Promise<HTMLImageElement> {
  1182→  1182→  return new Promise((resolve, reject) => {
  1183→  1183→    const img = new Image()
  1184→  1184→    img.crossOrigin = 'anonymous'
  1185→  1185→    img.onload = () => resolve(img)
  1186→  1186→    img.onerror = reject
  1187→  1187→    img.src = src
  1188→  1188→  })
  1189→  1189→}
  1190→  1190→
  1191→  1191→/**
  1192→  1192→ * Load a video file and wait until it has metadata + at least one frame
  1193→  1193→ * ready to draw. Used by the export pipeline to render MP4 video
  1194→  1194→ * backgrounds frame-by-frame onto the canvas. The video is muted and
  1195→  1195→ * paused — we drive it manually via `currentTime` seeking in the render
  1196→  1196→ * loop. `playsInline` is set so iOS Safari allows the metadata load
  1197→  1197→ * without a user gesture.
  1198→  1198→ */
  1199→  1199→function loadVideo(src: string): Promise<HTMLVideoElement> {
  1200→  1200→  return new Promise((resolve, reject) => {
  1201→  1201→    const v = document.createElement('video')
  1202→  1202→    v.src = src
  1203→  1203→    v.muted = true
  1204→  1204→    v.playsInline = true
  1205→  1205→    v.preload = 'auto'
  1206→  1206→    v.crossOrigin = 'anonymous'
  1207→  1207→    // Some browsers won't decode frames until the video starts playing
  1208→  1208→    // or at least has been "loadeddata". We wait for `loadeddata` (one
  1209→  1209→    // frame is available for drawing).
  1210→  1210→    v.onloadeddata = () => {
  1211→  1211→      // Try to start playback so the decoder actually decodes frames —
  1212→  1212→      // we immediately pause so there's no audible sound. This is the
  1213→  1213→      // most reliable cross-browser way to ensure `drawImage(video)`
  1214→  1214→      // produces a non-blank frame.
  1215→  1215→      v.play()
  1216→  1216→        .then(() => {
  1217→  1217→          v.pause()
  1218→  1218→          resolve(v)
  1219→  1219→        })
  1220→  1220→        .catch(() => {
  1221→  1221→          // Autoplay was blocked — fall back to a seeked-frame approach.
  1222→  1222→          // The video element should still be drawable after loadeddata
  1223→  1223→          // on most browsers (Chrome, Firefox, Safari desktop).
  1224→  1224→          resolve(v)
  1225→  1225→        })
  1226→  1226→    }
  1227→  1227→    v.onerror = () => reject(new Error(`Failed to load video: ${src}`))
  1228→  1228→    // Trigger the metadata load.
  1229→  1229→    v.load()
  1230→  1230→  })
  1231→  1231→}

  1270→  1270→function drawFrame({
  1271→  1271→  ctx,
  1272→  1272→  W,
  1273→  1273→  H,
  1274→  1274→  slide,
  1275→  1275→  intoMs,
  1276→  1276→  settings,
  1277→  1277→  bgImg,
  1278→  1278→  bgVideo,
  1279→  1279→  ayatIndex,
  1280→  1280→  total,
  1281→  1281→  attributionLine,
  1282→  1282→  reciterName,
  1283→  1283→  watermarkImg,
  1284→  1284→  splitPart,
  1285→  1285→}: DrawArgs) {
  1286→  1286→  // ---- Background (cover-fit) -----------------------------------------
  1287→  1287→  // For image backgrounds: draw the loaded <img> cover-fit.
  1288→  1288→  // For video backgrounds: draw the current frame of the loaded <video>
  1289→  1289→  //   (the render loop has already seeked it to the right time).
  1290→  1290→  // Fallback: solid dark color if neither is available.
  1291→  1291→  if (bgImg) {
  1292→  1292→    const ratio = Math.max(W / bgImg.width, H / bgImg.height)
  1293→  1293→    const dw = bgImg.width * ratio
  1294→  1294→    const dh = bgImg.height * ratio
  1295→  1295→    ctx.drawImage(bgImg, (W - dw) / 2, (H - dh) / 2, dw, dh)
  1296→  1296→  } else if (bgVideo && bgVideo.videoWidth > 0) {
  1297→  1297→    // Cover-fit: scale so the video fills the canvas, cropping overflow.
  1298→  1298→    const vw = bgVideo.videoWidth
  1299→  1299→    const vh = bgVideo.videoHeight
  1300→  1300→    const ratio = Math.max(W / vw, H / vh)
  1301→  1301→    const dw = vw * ratio
  1302→  1302→    const dh = vh * ratio
  1303→  1303→    ctx.drawImage(bgVideo, (W - dw) / 2, (H - dh) / 2, dw, dh)
  1304→  1304→  } else {
  1305→  1305→    ctx.fillStyle = '#0a0f1a'
  1306→  1306→    ctx.fillRect(0, 0, W, H)
  1307→  1307→  }
  1308→  1308→
  1309→  1309→  // ---- User overlay (using the selected preset shape) -----------------
  1310→  1310→  paintOverlayOnCanvas(ctx, W, H, settings)
  1311→  1311→
  1312→  1312→  // ---- Subtle top + bottom gradient for legibility (always on) --------
  1313→  1313→  const grad = ctx.createLinearGradient(0, 0, 0, H)

  1678→  1678→    // Build wrapped lines from the visible Tajweed words only.
  1679→  1679→    const tajweedLines: { text: string; color: string | null }[][] = []
  1680→  1680→    let curLine: { text: string; color: string | null }[] = []
  1681→  1681→    let curLineW = 0
  1682→  1682→    for (const wc of visibleTajweedWords) {
  1683→  1683→      const ww = ctx.measureText(wc.text).width
  1684→  1684→      if (curLineW + ww > innerMaxW && curLine.length > 0) {
  1685→  1685→        tajweedLines.push(curLine)
  1686→  1686→        curLine = []
  1687→  1687→        curLineW = 0
  1688→  1688→      }
  1689→  1689→      curLine.push(wc)
  1690→  1690→      curLineW += ww
  1691→  1691→    }
  1692→  1692→    if (curLine.length > 0) tajweedLines.push(curLine)
  1693→  1693→
  1694→  1694→    const visibleTajweedLines = tajweedLines
  1695→  1695→
  1696→  1696→    for (const lineSegs of visibleTajweedLines) {
  1697→  1697→      let totalW = 0
  1698→  1698→      for (const seg of lineSegs) totalW += ctx.measureText(seg.text).width
  1699→  1699→      let xPos = W / 2 + totalW / 2
  1700→  1700→      for (const seg of lineSegs) {
  1701→  1701→        const sw = ctx.measureText(seg.text).width
  1702→  1702→        xPos -= sw / 2
  1703→  1703→        ctx.fillStyle = seg.color ?? settings.fontColor
  1704→  1704→        ctx.shadowColor = 'rgba(0,0,0,0.9)'
  1705→  1705→        ctx.shadowBlur = H * 0.012
  1706→  1706→        ctx.fillText(seg.text, xPos, y)
  1707→  1707→        ctx.shadowBlur = 0
  1708→  1708→        xPos -= sw / 2
  1709→  1709→      }
  1710→  1710→      y += arabicLineH
  1711→  1711→    }
  1712→  1712→  } else {
  1713→  1713→    // ─── Plain word-by-word rendering ──────────────────────────────
  1714→  1714→    // `visibleArabicLines` already contains only the current chunk's
  1715→  1715→    // wrapped words. The highlight index `activeIdx` is a GLOBAL word
  1716→  1716→    // index (into the full ayat), so we subtract `visibleWordOffset`
  1717→  1717→    // to get the local index within the visible chunk.
  1718→  1718→    let wordCounter = 0
  1719→  1719→    for (const ln of visibleArabicLines) {
  1720→  1720→      const widths = ln.map((w) => ctx.measureText(w).width)
  1721→  1721→      const totalLineW = widths.reduce((a, b) => a + b, 0) + spaceW * (ln.length - 1)
  1722→  1722→      let xPos = W / 2 + totalLineW / 2
  1723→  1723→      for (let i = 0; i < ln.length; i++) {
  1724→  1724→        const w = ln[i]!
  1725→  1725→        const ww = widths[i]!
  1726→  1726→        xPos -= ww / 2
  1727→  1727→        // `activeIdx` is a GLOBAL word index across the whole ayat.
  1728→  1728→        // The visible chunk starts at `visibleWordOffset`, so the
  1729→  1729→        // local position of the active word (if any) inside this
  1730→  1730→        // chunk is `activeIdx - visibleWordOffset`. Only highlight
  1731→  1731→        // when that matches the local counter.
  1732→  1732→        const localActive = activeIdx - visibleWordOffset
  1733→  1733→        const isHi = wordCounter === localActive
  1734→  1734→        ctx.fillStyle = isHi ? settings.highlightColor : settings.fontColor
  1735→  1735→        if (isHi) {
  1736→  1736→          ctx.shadowColor = settings.highlightColor
  1737→  1737→          ctx.shadowBlur = H * 0.028
  1738→  1738→        } else {
  1739→  1739→          ctx.shadowColor = 'rgba(0,0,0,0.9)'
  1740→  1740→          ctx.shadowBlur = H * 0.012
  1741→  1741→        }
  1742→  1742→        ctx.fillText(w, xPos, y)
  1743→  1743→        ctx.shadowBlur = 0
  1744→  1744→        xPos -= ww / 2 + spaceW
  1745→  1745→        wordCounter++
  1746→  1746→      }
  1747→  1747→      y += arabicLineH
  1748→  1748→    }
  1749→  1749→  }
  1750→  1750→
  1751→  1751→  // After Arabic loop, y is at the bottom of the last Arabic line.
  1752→  1752→  // Reset y to a clean position based on the card layout, not the loop variable.
  1753→  1753→  // This prevents overlap when the Arabic loop's y drifts.
  1754→  1754→
  1755→  1755→  // Transliteration
  1756→  1756→  let yAfterArabic = cardY + cardPadY + arabicTotalH
  1757→  1757→  if (translit) {
  1758→  1758→    yAfterArabic += arabicToTranslitGap
  1759→  1759→    ctx.font = `italic ${translitFontSize}px Inter, sans-serif`
  1760→  1760→    ctx.fillStyle = 'rgba(255,255,255,0.72)'
  1761→  1761→    ctx.textAlign = 'center'
  1762→  1762→    ctx.textBaseline = 'middle'
  1763→  1763→    ctx.fillText(translit, W / 2, yAfterArabic + translitFontSize / 2)
  1764→  1764→    yAfterArabic += translitH
  1765→  1765→  }
  1766→  1766→
  1767→  1767→  // Divider
  1768→  1768→  if (dividerGap) {
  1769→  1769→    yAfterArabic += Math.round(H * 0.008)
  1770→  1770→    ctx.strokeStyle = hexWithAlpha(settings.fontColor, 0.4)
  1771→  1771→    ctx.lineWidth = 1
  1772→  1772→    ctx.beginPath()
  1773→  1773→    ctx.moveTo(W / 2 - W * 0.03, yAfterArabic)
  1774→  1774→    ctx.lineTo(W / 2 + W * 0.03, yAfterArabic)
  1775→  1775→    ctx.stroke()
  1776→  1776→    yAfterArabic += Math.round(H * 0.008)
  1777→  1777→  }
  1778→  1778→
  1779→  1779→  // Translation — use the clean yAfterArabic position, not the loop's y
  1780→  1780→  if (transLines.length) {
  1781→  1781→    if (!dividerGap) yAfterArabic += arabicToTransGap
  1782→  1782→    ctx.font = `${transFontSize}px ${transFontFamily}`
  1783→  1783→    ctx.fillStyle = 'rgba(255,255,255,0.88)'
  1784→  1784→    ctx.textAlign = 'center'
  1785→  1785→    ctx.textBaseline = 'middle'
  1786→  1786→    ctx.shadowColor = 'rgba(0,0,0,0.9)'
  1787→  1787→    ctx.shadowBlur = H * 0.008
  1788→  1788→    for (let i = 0; i < transLines.length; i++) {
  1789→  1789→      ctx.fillText(transLines[i]!, W / 2, yAfterArabic + transLineH / 2 + i * transLineH)
  1790→  1790→    }
  1791→  1791→    ctx.shadowBlur = 0
  1792→  1792→  }
  1793→  1793→
  1794→  1794→  // ─── Page indicator ───────────────────────────────────────────────
  1795→  1795→  // If either Arabic or translation is paginated, show a small "1/3" indicator
  1796→  1796→  // below the translation so viewers know there's more text coming.
  1797→  1797→  if (arabicPageInfo || transPageInfo) {
  1798→  1798→    const pageInfo = transPageInfo || arabicPageInfo!
  1799→  1799→    const indicatorFontSize = Math.round(H * 0.012)
  1800→  1800→    ctx.font = `${indicatorFontSize}px Inter, sans-serif`
  1801→  1801→    ctx.fillStyle = 'rgba(255,255,255,0.35)'
  1802→  1802→    ctx.textAlign = 'center'
  1803→  1803→    ctx.textBaseline = 'middle'
  1804→  1804→    const indicatorY = yAfterArabic + transTotalH + Math.round(H * 0.015)
  1805→  1805→    ctx.fillText(`${pageInfo.current} / ${pageInfo.total}`, W / 2, indicatorY)
  1806→  1806→  }
  1807→  1807→
  1808→  1808→  // ---- Attribution block (bottom-left) --------------------------------
  1809→  1809→  // Shows the translation attribution (when required) + the reciter credit
  1810→  1810→  // (always). Stacked vertically so both are visible.
  1811→  1811→  const attrFontSize = Math.round(H * 0.016)
  1812→  1812→  const attrLineH = attrFontSize * 1.35
  1813→  1813→  const attrBottom = H * 0.985
  1814→  1814→  const maxAttrW = W * 0.55
  1815→  1815→
  1816→  1816→  // Helper: truncate text to fit within maxAttrW
  1817→  1817→  function truncateAttr(text: string): string {
  1818→  1818→    let t = text
  1819→  1819→    while (ctx.measureText(t).width > maxAttrW && t.length > 10) {
  1820→  1820→      t = t.slice(0, -2) + '…'
  1821→  1821→    }
  1822→  1822→    return t
  1823→  1823→  }
  1824→  1824→
  1825→  1825→  ctx.font = `${attrFontSize}px Inter, sans-serif`
  1826→  1826→  ctx.fillStyle = 'rgba(255,255,255,0.45)'
  1827→  1827→  ctx.textAlign = 'left'
  1828→  1828→  ctx.textBaseline = 'bottom'
  1829→  1829→
  1830→  1830→  // Draw reciter credit first (lower), then translation attribution (above it)
  1831→  1831→  if (reciterName) {
  1832→  1832→    ctx.fillText(`Recited by ${reciterName}`, W * 0.03, attrBottom)
  1833→  1833→  }
  1834→  1834→  if (attributionLine) {
  1835→  1835→    ctx.fillText(truncateAttr(attributionLine), W * 0.03, attrBottom - attrLineH)
  1836→  1836→  }
  1837→  1837→
  1838→  1838→  // ---- Structural Quran markers (Juz / Hizb / Rubʿ / Ruku / Manzil / Page) ----
  1839→  1839→  // Drawn at the bottom-center of the frame, just above the attribution
  1840→  1840→  // block. Only shows fields that are actually present on the verse.
  1841→  1841→  // Matches the live preview's formatting so WYSIWYG.
  1842→  1842→  {
  1843→  1843→    const pairs = getStructuralPairs(slide)
  1844→  1844→    if (pairs.length > 0) {
  1845→  1845→      const structText = formatStructural(slide, true) // uppercase for export
  1846→  1846→      const structFontSize = Math.round(H * 0.014)
  1847→  1847→      ctx.save()
  1848→  1848→      ctx.font = `500 ${structFontSize}px Inter, sans-serif`
  1849→  1849→      ctx.fillStyle = 'rgba(255,255,255,0.55)'
  1850→  1850→      ctx.textAlign = 'center'
  1851→  1851→      ctx.textBaseline = 'bottom'
  1852→  1852→      ctx.shadowColor = 'rgba(0,0,0,0.6)'
  1853→  1853→      ctx.shadowBlur = Math.max(1, Math.round(structFontSize * 0.2))
  1854→  1854→      // Letter-spacing for the uppercase tracking look — guarded for Safari.
  1855→  1855→      type CtxWithLs = CanvasRenderingContext2D & { letterSpacing?: string }
  1856→  1856→      const ctxLs = ctx as CtxWithLs
  1857→  1857→      if (typeof ctxLs.letterSpacing === 'string') {
  1858→  1858→        try {
  1859→  1859→          ctxLs.letterSpacing = `${Math.round(structFontSize * 0.08)}px`
  1860→  1860→        } catch {
  1861→  1861→          // ignore
  1862→  1862→        }
  1863→  1863→      }
  1864→  1864→      ctx.fillText(structText, W / 2, H * 0.965)
  1865→  1865→      ctx.restore()
  1866→  1866→    }
  1867→  1867→  }
  1868→  1868→
  1869→  1869→  // ---- Top-center watermark: Jariyah Now brand mark -------------------
  1870→  1870→  // Rendered at the TOP LEVEL of the composition (inside drawFrame, so it
  1871→  1871→  // appears on every frame, not tied to any single ayat's sequence).
  1872→  1872→  {
  1873→  1873→    const minDim = Math.min(W, H)
  1874→  1874→    const wmY = H * 0.04 // top: 4% from top edge
  1875→  1875→
  1876→  1876→    if (watermarkImg && watermarkImg.complete && watermarkImg.naturalWidth > 0) {
  1877→  1877→      // Image watermark — scale to a target height proportional to the canvas size.
  1878→  1878→      // 112px @ 720p (clearly visible but not overpowering), scales to ~168px @ 1080p.
  1879→  1879→      const targetH = Math.round((minDim / 720) * 112)
  1880→  1880→      const scale = targetH / watermarkImg.naturalHeight
  1881→  1881→      const targetW = Math.round(watermarkImg.naturalWidth * scale)
  1882→  1882→      const x = Math.round((W - targetW) / 2) // horizontally centered
  1883→  1883→
  1884→  1884→      ctx.save()
  1885→  1885→      ctx.globalAlpha = 0.9
  1886→  1886→      ctx.drawImage(watermarkImg, x, wmY, targetW, targetH)
  1887→  1887→      ctx.restore()
  1888→  1888→    } else {
  1889→  1889→      // Text fallback — used only if the watermark PNG hasn't loaded yet.
  1890→  1890→      const wmFontSize = Math.round((minDim / 720) * 14)
  1891→  1891→      ctx.save()
  1892→  1892→      ctx.font = `500 ${wmFontSize}px Inter, sans-serif`
  1893→  1893→      ctx.fillStyle = 'rgba(255, 255, 255, 0.75)'
  1894→  1894→      ctx.textAlign = 'center'
  1895→  1895→      ctx.textBaseline = 'top'
  1896→  1896→      ctx.shadowColor = 'rgba(0, 0, 0, 0.6)'
  1897→  1897→      ctx.shadowOffsetX = 0
  1898→  1898→      ctx.shadowOffsetY = Math.max(1, Math.round(wmFontSize * 0.07))
  1899→  1899→      ctx.shadowBlur = Math.round(wmFontSize * 0.28)
  1900→  1900→      ctx.fillText('Made with Jariyah Now', W / 2, wmY)
  1901→  1901→      ctx.restore()
  1902→  1902→    }
  1903→  1903→  }
  1904→  1904→}

  1906→  1906→function wrapLines(
  1907→  1907→  ctx: CanvasRenderingContext2D,
  1908→  1908→  text: string,
  1909→  1909→  maxW: number,
  1910→  1910→): string[] {
  1911→  1911→  const words = text.split(/\s+/)
  1912→  1912→  const lines: string[] = []
  1913→  1913→  let line = ''
  1914→  1914→  for (const w of words) {
  1915→  1915→    const test = line ? line + ' ' + w : w
  1916→  1916→    if (ctx.measureText(test).width > maxW && line) {
  1917→  1917→      lines.push(line)
  1918→  1918→      line = w
  1919→  1919→    } else {
  1920→  1920→      line = test
  1921→  1921→    }
  1922→  1922→  }
  1923→  1923→  if (line) lines.push(line)
  1924→  1924→  return lines
  1925→  1925→}

  1927→  1927→function hexWithAlpha(hex: string, alpha: number): string {
  1928→  1928→  const h = hex.replace('#', '')
  1929→  1929→  const r = parseInt(h.slice(0, 2), 16)
  1930→  1930→  const g = parseInt(h.slice(2, 4), 16)
  1931→  1931→  const b = parseInt(h.slice(4, 6), 16)
  1932→  1932→  return `rgba(${r}, ${g}, ${b}, ${alpha})`
  1933→  1933→}

  1935→  1935→function roundedRect(
  1936→  1936→  ctx: CanvasRenderingContext2D,
  1937→  1937→  x: number,
  1938→  1938→  y: number,
  1939→  1939→  w: number,
  1940→  1940→  h: number,
  1941→  1941→  r: number,
  1942→  1942→) {
  1943→  1943→  const radius = Math.min(r, w / 2, h / 2)
  1944→  1944→  ctx.beginPath()
  1945→  1945→  ctx.moveTo(x + radius, y)
  1946→  1946→  ctx.arcTo(x + w, y, x + w, y + h, radius)
  1947→  1947→  ctx.arcTo(x + w, y + h, x, y + h, radius)
  1948→  1948→  ctx.arcTo(x, y + h, x, y, radius)
  1949→  1949→  ctx.arcTo(x, y, x + w, y, radius)
  1950→  1950→  ctx.closePath()
  1951→  1951→}

  1953→  1953→function formatMs(ms: number): string {
  1954→  1954→  if (!Number.isFinite(ms) || ms <= 0) return '0:00'
  1955→  1955→  const total = Math.floor(ms / 1000)
  1956→  1956→  const m = Math.floor(total / 60)
  1957→  1957→  const s = total % 60
  1958→  1958→  return `${m}:${String(s).padStart(2, '0')}`
  1959→  1959→}
