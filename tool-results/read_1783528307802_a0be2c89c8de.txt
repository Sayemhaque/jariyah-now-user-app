    56тЖТ    56тЖТ// Split a text into word chunks for pagination. Used when an ayah is
    57тЖТ    57тЖТ// very long (e.g. Ayat al-Kursi) and the text would be cramped.
    58тЖТ    58тЖТfunction getWordChunk(text: string, chunkIdx: number, totalChunks: number): string {
    59тЖТ    59тЖТ  const words = text.split(/\s+/).filter(Boolean)
    60тЖТ    60тЖТ  if (totalChunks <= 1) return text
    61тЖТ    61тЖТ  const perChunk = Math.ceil(words.length / totalChunks)
    62тЖТ    62тЖТ  const start = chunkIdx * perChunk
    63тЖТ    63тЖТ  const end = start + perChunk
    64тЖТ    64тЖТ  return words.slice(start, end).join(' ')
    65тЖТ    65тЖТ}

    67тЖТ    67тЖТfunction calcTotalChunks(arabicText: string, translation: string): number {
    68тЖТ    68тЖТ  const arWords = arabicText.split(/\s+/).filter(Boolean).length
    69тЖТ    69тЖТ  const transWords = translation.split(/\s+/).filter(Boolean).length
    70тЖТ    70тЖТ  const arChunks = arWords > 20 ? Math.ceil(arWords / 8) : 1
    71тЖТ    71тЖТ  const transChunks = transWords > 24 ? Math.ceil(transWords / 12) : 1
    72тЖТ    72тЖТ  return Math.max(arChunks, transChunks, 1)
    73тЖТ    73тЖТ}
