    69→// A flat list of renderable Arabic tokens — either individual words
    70→// (with optional highlight index) or color-coded Tajweed fragments.
    71→interface ArabicToken {
    72→  text: string
    73→  color?: string | null // null = use default font color
    74→  wordIdx?: number // global word index (used for highlight matching)
    75→}
    76→
    77→function buildArabicTokens(current: {
    78→  words: { text: string }[]
    79→  tajweedSegments?: { text: string; color: string | null }[] | null
    80→  arabicText: string
    81→}): ArabicToken[] {
    82→  if (current.words.length > 0) {
    83→    return current.words.map((w, i) => ({ text: w.text, wordIdx: i }))
    84→  }
    85→  if (current.tajweedSegments && current.tajweedSegments.length > 0) {
    86→    // Flatten Tajweed segments into individual words while preserving
    87→    // the per-segment color. Splits on whitespace but keeps the color
    88→    // attached to each resulting word.
    89→    const tokens: ArabicToken[] = []
    90→    for (const seg of current.tajweedSegments) {
    91→      for (const part of seg.text.split(/\s+/)) {
    92→        if (part) tokens.push({ text: part, color: seg.color })
    93→      }
    94→    }
    95→    return tokens
    96→  }
    97→  if (current.arabicText) {
    98→    return current.arabicText
    99→      .split(/\s+/)
   100→      .filter(Boolean)
   101→      .map((text, i) => ({ text, wordIdx: i }))
   102→  }
   103→  return []
   104→}

   106→/**
   107→ * Build the pagination plan for the current ayat. Prefers silence-snapped
   108→ * boundaries (from ffmpeg pause detection on the ACTUAL audio) and falls
   109→ * back to time-proportional when pauses aren't available.
   110→ *
   111→ * We do NOT use the word-timing API (buildAudioSyncedChunks) because
   112→ * the reciter IDs don't match between UmmahAPI and quran.com, and
   113→ * even if they did, isolated word MP3 durations don't match continuous
   114→ * recitation timings.
   115→ */
   116→function buildPlanForAyat(current: {
   117→  words: { text: string; startMs: number; endMs: number }[]
   118→  arabicText: string
   119→  translation: string
   120→  audioDurationMs: number
   121→  audioPauses?: { start: number; end: number; duration: number }[]
   122→}): PaginationPlan {
   123→  const arWords = current.arabicText.split(/\s+/).filter(Boolean).length
   124→  const n = estimateChunkCount(arWords)
   125→  if (n <= 1 || current.audioDurationMs <= 0) {
   126→    return { chunks: [], total: 0 }
   127→  }
   128→  // ✅ Best: silence-snapped from the real audio's breath pauses
   129→  if (current.audioPauses && current.audioPauses.length > 0) {
   130→    return buildSilenceSnappedPlan(
   131→      current.audioDurationMs,
   132→      n,
   133→      current.audioPauses,
   134→      arWords,
   135→    )
   136→  }
   137→  // Fallback: even time split
   138→  return buildTimeProportionalPlan(current.audioDurationMs, n)
   139→}

   249→  // Use a ref to break the recursive self-reference inside `tick` — this
   250→  // also lets us always read the latest `current`/`currentIndex` without

   252→  const stateRef = useRef({ current, currentIndex })
   253→  useEffect(() => {
   254→    stateRef.current = { current, currentIndex }

   258→    if (!isPlaying) return

   260→

   262→      if (cancelled) return

   264→      const { current: c, currentIndex: ci } = stateRef.current

   269→        // Build the chunk plan for this ayat (cheap; pure function of
   270→        // the ayat's word timings + duration). Then find which chunk
   271→        // is active at the current audio position.
   272→        //
   273→        // We DON'T memoize here because the tick runs every frame and
   274→        // we want zero stale closures — `buildPlanForAyat` is fast
   275→        // enough (a few hundred microseconds for a 50-word ayat).
   276→        const dur = c.audioDurationMs || liveDurations[ci] || 0

   278→          words: c.words,
   279→          arabicText: c.arabicText,
   280→          translation: c.translation,
   281→          audioDurationMs: dur,
   282→          audioPauses: c.audioPauses,

   288→          // a needless React re-render every animation frame.
   289→          setTextPage((prev) => (prev === page ? prev : page))
   290→        } else if (plan.total <= 1) {
   291→          setTextPage((prev) => (prev === 0 ? prev : 0))
   292→        }

   294→        if (foundIdx >= 0) {

   296→        }

   299→    }

   321→        // Autoplay can fail; show paused state and let the user retry.
   322→        setIsPlaying(false)
   323→      })

   328→  // ─── Preload the NEXT ayat's MP3 in parallel ─────────────────────────

   330→  // the next ayat's MP3. The browser fetches + decodes it in the
   331→  // background. When the current ayat ends, we swap the preloaded src
   332→  // into the main audio element — playback resumes in ~50ms instead of
   333→  // the ~1000ms gap caused by fetching on demand.
   334→  useEffect(() => {
   335→    if (!ayatList.length) return
   336→    const nextIdx = currentIndex + 1
   337→    if (nextIdx >= ayatList.length) return // last ayat — nothing to preload
   338→    const nextUrl = ayatList[nextIdx]?.audioUrl
   339→    if (!nextUrl) return
   340→
   341→    // Reuse the same hidden Audio() instance, just swap the src.
   342→    // `preload='auto'` tells the browser to fetch + buffer aggressively.
   343→    if (!nextAudioRef.current) {
   344→      nextAudioRef.current = new Audio()
   345→    }
   346→    const nextAudio = nextAudioRef.current
   347→    // Skip if already preloading/loaded this exact URL (avoid re-fetch).
   348→    if (nextAudio.src !== nextUrl) {
   349→      nextAudio.src = nextUrl
   350→      nextAudio.preload = 'auto'
   351→      // Some browsers won't actually fetch until play() is called.
   352→      // Calling play() then immediately pausing forces the preload
   353→      // without producing audible sound.
   354→      nextAudio.muted = true
   355→      nextAudio.volume = 0
   356→      nextAudio.play().then(() => {
   357→        nextAudio.pause()
   358→        nextAudio.currentTime = 0
   359→        nextAudio.muted = false
   360→        nextAudio.volume = 1
   361→      }).catch(() => {
   362→        // Autoplay policy blocked the preload — fall back to setting
   363→        // preload='auto' and letting the browser fetch metadata-only.
   364→        nextAudio.muted = false
   365→        nextAudio.volume = 1
   366→      })
   367→    }
   368→  }, [currentIndex, ayatList])
   369→
   370→  // Cleanup: release the preloader Audio() when the component unmounts
   371→  // so it doesn't keep a network connection open in the background.
   372→  useEffect(() => {
   373→    return () => {
   374→      if (nextAudioRef.current) {
   375→        nextAudioRef.current.src = ''
   376→        nextAudioRef.current = null
   377→      }
   378→    }
   379→  }, [])

   384→    if (!audio) return

   498→  // Pre-probe audio durations for all ayats when the list loads.
   499→  // The store's getAudioDurationMs may return 0 under COEP — this is a
   500→  // second attempt using the actual audio element in the DOM. We load each
   501→  // ayat's audio URL silently (muted, not playing) just to get the metadata.

   503→    if (!ayatList.length) return

   506→      if (ayat.audioDurationMs > 0) return // already have it

   508→      a.preload = 'metadata'
   509→      a.crossOrigin = 'anonymous'
   510→      a.onloadedmetadata = () => {
   511→        const d = a.duration
   512→        if (isFinite(d) && d > 0) {
   513→          setLiveDurations((prev) => ({ ...prev, [i]: d * 1000 }))
   514→        }
   515→      }
   516→      a.src = ayat.audioUrl
   517→      probes.push(a)
   518→    })
   519→    return () => {
   520→      probes.forEach((a) => { a.src = '' })
   521→    }
   522→  }, [ayatList])

   524→  // Build the overlay background expression from the user's chosen preset.
   525→  const overlayBg = overlayCssBackground(settings)

   530→  // `background-image`. The export pipeline (ExportModal.tsx) detects the

   532→  const isVideoBg = settings.backgroundImage.endsWith('.mp4')

   534→  // Find the word to highlight in the current ayat

   536→    activeWord && activeWord.ayatIndex === currentIndex
   537→      ? activeWord.wordIndex
   538→      : -1

   540→  // ─── Memoized audio-synced pagination plan for the CURRENT ayat ────
   541→  // Built once per ayat (or per duration change). The rAF tick uses
   542→  // the same plan to decide which chunk is active at `audio.currentTime`.
   543→  // This drives the live preview's text pagination — chunk N switches
   544→  // the instant the reciter finishes the last word in chunk N.
   545→  const paginationPlan = useMemo<PaginationPlan>(() => {
   546→    if (!current) return { chunks: [], total: 0 }

  1048→function formatMs(ms: number): string {
  1049→  if (!Number.isFinite(ms) || ms <= 0) return '0:00'
  1050→  const total = Math.floor(ms / 1000)
  1051→  const m = Math.floor(total / 60)
  1052→  const s = total % 60
  1053→  return `${m}:${String(s).padStart(2, '0')}`
  1054→}
